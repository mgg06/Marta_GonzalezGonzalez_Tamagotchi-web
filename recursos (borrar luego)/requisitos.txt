/* Ejemplo de Colisión (Especificidad) */
/* Un selector de etiqueta (baja especificidad) */
p {
color: red;
}
/* Un selector de clase (alta especificidad) GANA al de etiqueta */
.texto-verde {
color: green;
}
<!-- Este párrafo será verde, porque la clase .texto-verde es más
específica -->
<p class="texto-verde">Texto de ejemplo</p>


/* 5. Selector de Atributo ([attr]) */
input[type="email"] {
border-color: blue; /* Estilo solo para los inputs de tipo email */
}
